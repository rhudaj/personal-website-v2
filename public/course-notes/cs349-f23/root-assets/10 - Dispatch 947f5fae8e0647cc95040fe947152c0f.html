    function buildTargetRoute(mx, my, target: SKElement) {
	route = [] 
	// RECURSIVE CASE -> SUBTREE
	if(target instanceof SKContainer) {
		target.children.forEach((child) => 
			buildTargetRoute(mx - target.x, my - target.y, child)
	}
	// BASE CASE -> This was the target that was hit
	if(element.hittest(mx,my) return [element, ...route]
	else return route 
}

3. Event Propagation

Event binding: adding/calling a handler to a widget for specific events

2 Phases

  • Events go up and down, widget can use the event either time with a handler
  • The handler can also stop propagation

  • Phase 1 ‚Äî capture
    • down the route
  • Phase 2 ‚Äî bubble
    • up through parents to root
